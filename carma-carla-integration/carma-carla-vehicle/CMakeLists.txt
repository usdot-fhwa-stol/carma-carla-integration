cmake_minimum_required(VERSION 2.8.3)
project(carla-carma-vehicle)

find_package(catkin REQUIRED COMPONENTS rospy roslaunch rostest)

#catkin_python_setup()

roslaunch_add_file_check(launch IGNORE_UNSET_ARGS)

catkin_package(CATKIN_DEPENDS rospy)

catkin_install_python(PROGRAMS src/carla_carma_vehicle/carla_carma_vehicle.py
                      DESTINATION ${CATKIN_PACKAGE_BIN_DESTINATION})

install(DIRECTORY launch/
        DESTINATION ${CATKIN_PACKAGE_SHARE_DESTINATION}/launch)

install(DIRECTORY config/
        DESTINATION ${CATKIN_PACKAGE_SHARE_DESTINATION}/config)

catkin_add_nosetests(test/test_carla-carma-vehicle.py)
